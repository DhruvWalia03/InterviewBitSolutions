/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

vector<int> Solution::inorderTraversal(TreeNode* A) {
    
    vector<int> ans;
    stack<TreeNode*> s;
    while(!s.empty() || A != NULL)
    {
        while(A != NULL)
        {
            ans.push_back(A->val);
            s.push(A);
            A = A->left;
        }
        A = s.top();
        s.pop();
        A = A->right;
    }
    return ans;
}
