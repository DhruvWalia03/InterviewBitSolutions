int Solution::solve(int A, int B, int C, int D) {
 
    if(A == B && C == D && A != C)
        return 1;
    if(A == C && B == D && A != B)
        return 1;
    if(A == D && C == B && A != C)
        return 1;
    return 0;
    
}
